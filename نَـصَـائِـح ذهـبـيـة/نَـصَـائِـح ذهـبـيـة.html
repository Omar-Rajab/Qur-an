قال الشافعـى رحمـه اللـه:
ولا تُعطين الرَّأى من لا يريده، فلا أنت محـمود ولا الـرَّأى نافِـعُـه.

عـلـمـت أن رزقـى فـي السـمـاء لا يـأخـذه غـيـرى فـاطـمـئن قـلبـى

يـجـب أن تعـلـم أن الـرزّاق هـو اللـه وحـده ومـاهـو ليـس لـك فهـو لـغـيـرك ومـاهـو لـك لا يمـكـن أن يـكـون لغـيـرك

إياك وترك الصلاة

إياك أن تـتـرك صـلاتـك مهـمـا كانـت ظـروفـك وإيـاك أن تسـمـح لأحـد أن يُـعـجِـزُك فالـصـلاة هـى أول مـا يُـسـأل عـنـه العـبـد يـوم القـيـامـة فـقـال اللـه تعـالـى فـى كتـابـه الكـريـم
مَـا سَـلَـكَـكُـم فـىِ سَـقَـرَ * قَـالُـوا لَـم نَـكُ مِـنَ المُـصَـلِّـينَ (المدثر : 42 - 43)

الـشَّـكَّ فـى النَّـفـس

إياك وألـشـك فـى نَـفـسِـك فهـذا مـن الشـيـطـان فـإنَّـه يُهـيِّـئ للـعـبـدِ أنَـهُ إذا أذنـب لـن يَغـفِـرَ اللـه لـه فاللـه كـمـا هـو شـديـد العِـقـاب فـهـو غـفـورٌ رحـيـم

صـاحِـب مَـن يأخُـذَكَ للجـنـه

حـاول أن تبـتـعد عـن الصُـحـبه الـسـوء فـمـا هـو سـوء سـيـؤدى إلـى سـوء

الـقـرآن طـارد للـحـزن

إذا شـعـرت يـومـاً بالـحُـزن وضـيـق فـى صـدرك فاذهـب ورتِّـل الـقـرآن فـهـو شِـفـاءٌ للنـفـس وطـارد للـحُـزن ومُذيب الهـمـوم

القـرآن العـظـيـم أصـل السَّـعـادة

إجعل لك وِرد يومى من القرآن والذِّكر واعتاد أن يكون لسانك رطباً بِذِكرِ الله

ألـا بـذِكِـرِ اللـه تَـطـمَـئـنُّ الـقـلـوب

إذا شـعـرت يـومـاً بالـخَـوفِ فـاذكُـرِ اللـه فـوللـه إن القـلـوبَ مُحَـصّـنـةٌ بِـذِكـرِ ربِّـهـا

لـو كـان حـدث هـذا لـفـعـلـتُ ذاك

إيـاك وأن تتـمـنـى قـدراً غـيـر قـدرِك فـيـجـب عـلـيـك أن تـعـتاد عـلـى حـمـد رَبِّـك وستـكـونُ مـن الفـائِـزيـن

مـاذا لـو عـاد مـعـتـذِراً ؟!

حـاول أن تـعـتاد عـلـى السـمـاح وأن تُـطـهِّـر قَلـبَـكَ مـن القـسـوةِ فـمـا نـحـن فـى هـذه الـدُّنـيـا إلا عـابـرى سـبـيـل فـاتـروكـهـا بِـسـلام حـتـى تَـسـلَـمـوا

إيـاك والشـكـوى لأصـحـاب القـبـور

إتـركـوا أصـحـاب الـقـبـور فـهـم ذهـبـوا إلـى عـمـلـهـم فهم لا يـسـمـعـونَـكُـم ولا يـرونَـكُـم، فـإذا اردت الشـكـوى فاسـتـجُـد بـيـن يـدى اللـه واشـتـكـى إلـيه فـهـو حَـسـبُـنا وَوَكـيـلُـنـا

بِـرُّ الـوالـديـنِ حـقٌ عـليـنا

إيـاك وأن تـغـضـب والـديـك يـومـاً فـكـما جـعـلـوا مـنـك أمـيـراً فـى صِـغَـرِك فـأجعـلـهـم مـلـوكـاً فـى كِـبَـرِهِـم

أهـل بـيـتِـكَ أمـانـتـك

يجـب أن تـعـلم أن زوجـتـك - زوجـك هـو أمانـة وهـذه الأمـانـة سـتُـسـأل عـنـهـا يـوم القـيـامة فـاتَّـقِ اللـه فى أهـل بـيـتـك وافـعـل مـا امـرك بـه اللـه ولا تـخُـن الأمـانـه

لا يـجـب عـلـى المُـسـلـم أن يـيـأس واللـه ربُّـه

لا تيـأس مـن شـئ فـى هـذه الـدنـيـا فكـل شـئ وارد حـدوثـه فهـنـاك مـن يـولـدُ فـقـيـراً ويـمـوت أمـيـراً وهـنـاك مـن يـولـدُ أمـيـراً ويـمـوتُ فـقـيـراً، والمكـتـوب لك فهو لـك والمـكـتـوب لـغـيـرك فـهـو لغـيـرك

إيـاك وحُـبُ المـال والشَّـهـوة

إن حـب المـال والشـهـوة هـو بـدايـة طـريـق الـهـلاك ولا تنـسـى أن المـال هـو فـقـط وسـيـلـة لتـحـقيـق شـئ آخـر فـلا تـضـع نفـسـك تـحـت رحـمـة الـمـال والشـهـوة

إتَّـبِـعِ الـحـقَّ واتـرُكِ البـاطـل

اذا شـهـدت فـاشـهـد بالـحـق ولـو على أُمِّـكَ وأبـيـك فـإنَّ اللـه لا يـسـتـحـى مـن الـحـق

لا تـكُـن مـن أصـحـاب الـرشـوة

إبـتـعـد عـن الـرَّشـوة فـإنَّـهـا مُهـلِـكَـة لـصـاحـبـهـا ولـقـد لَـعَـنَ رسـول اللـه صـلـى اللـه عـلـيـه وسـلـم الـرَّاشـى والمُـرتـشـي

عـلاقـة الـزَّوج بـزوجـتـه

إن الزوجـة هـى مَـأمَـن زوجـهـا والـزوجُ مَـأمَـن زوجتـه والعـلاقـة بـيـن الـرجـل والـمـرأه هـى عـلاقـة تـبـادلـيـة أى إذا فـعـلـت سـتـجـد نِـتـاج مـا فـعـلـتـه سـواء كـان خـيـر أم شـر فـأحـسـنوا إلـى أزوجِـكـم

تـزوَّج بـمـن أردت لا بـمـن يريد لَـك

إذا نَـويـتُ أن تـتـزوَّج فـتـزوَّج بـمـن أردت ورضـيـت أن يـكـون زوجـاً لـك وأحـسِـنـوا الإخـتـيـار حتى لا يلحقُكُمُ النَّدَم

إتقوا الله فى أولادِكُم

إن أولادِكُـم أمـانـة فـربُّـوا أولادِكُـم تـربـيـةً حـسـنـةً تـرضـى اللـه ورسـولـه فـإن هـؤلاء الأبـنـاء سـيـسـتـغـفـرون لكـم بـعـد وفـاتِـكُـم

إياك وسب الدِّين

إيـاك ثـم إيـاك ثـم إيـاك أن تَـسُـب ديـن اللـه سـواء كـان جـادَّاً أو مـازِحـاً فتـكـون مـن أهـل الكُـفـر

الغـضـب فـى غَـيـرَ مَـحـلّـه مُـدَمِّـر

لا تجـعـل الغـضـب يـتـملَّـك مـنـك حـتـى لا يُهـلِكَـكَ فتـعـلـم كـيـف تـعـتاد عـلـى امـتصـاص الغـضـب ولا تغـضـب إلا لـديـن اللـه

فى الحُبِ دُعـاء

إذا أحـبـبـتُ شـخـصـاً مـا خـذه مـعـك فـى دُعـائك دون علـمـه، فهـكـذا يـكـون الـحـب أجـمـل، أصـفـى، أنـقـى، وأقـرب إلـى اللـه

حـسـن الـظـن باللـه غِـنـاء

كُـن عـلـى يقـيـن أن اللـه هو الـذى يعطـيـك وهـو الـذى يـأخـذُ مـنـك وكُلّ شـئٍ بِحِـكـمـة، وإذا ذهـب الجـمـيـل سـيأتى ماهو أجـمـل

نعـمـة السـتـر نـعـمـة مـجـهـولـه

كثـيـر مـن النَّـاس لا يـدرك نـعـمـة السـتر ولا يـدرك أننـا تـحـت غـطـاء السـتر مـالـم يفضحـنـا اللـه
إذا اردت أن تدعـو اللـه بالسـتـر فـالأولَـى أن تـدعـوه بـدوام نعـمـة السـتـر لأنـه يسـتـرك ولـكـنـك لا تـدرى، اللـهـم أدِم علـيـنا سـتـرك الجـمـيـل واجـعـلـنا خـلـف السـتـر مايُـرضـيـك
